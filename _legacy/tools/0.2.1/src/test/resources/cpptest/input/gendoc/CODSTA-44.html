Local variables and variables of class/parent classes/parent structs should have different name [CODSTA-44-1]

DESCRIPTION 

Rule checks if local variables and variables of class/parent classes/
/parent structs have the same name (case sensitive). 
If so, then it reports violation message.
See also: CODSTA-45
BENEFITS 

Rule improves readability and prevents writing ambiguous code.
EXAMPLE 

class Base {
public:
    int member;
    float var1;
    float var2;
    short var3;

    void foo( ) {
        float member;                                   // Violation
    }

    short foo2( ) {
        for( int var2 = 0; var2 < 10; var2++ ) {  // Violation
        }
        return var3;
    }
};


class A : public Base {
    void foo( ) {
        float member;                                   // Violation
    }
    int foo1( );
};
REPAIR 

class Base {
public:
    int member;
    float var1;
    float var2;
    short var3;

    void foo( ) {
        float member2;                                     // OK
    }

    short foo2( ) {
        for( int var4 = 0; var4 < 10; var4++ ) {     // OK
        }
        return var3;
    }
};

class A : public Base {
    void foo( ) {
        float member2;                                     // OK
    }
    int foo1( );
};
